<script>
    const raceReq = { 1:14, 2:19, 3:25, 4:31, 5:38, 6:46, 7:55, 8:65, 9:75 };

    function getMatchPoints(score, target, oppScore, oppTarget) {
        if (score >= target) return 20;
        if (oppScore >= oppTarget) {
            const pct = (score / target) * 100;
            if (pct >= 90) return 8; if (pct >= 80) return 7; if (pct >= 70) return 6;
            if (pct >= 60) return 5; if (pct >= 50) return 4; if (pct >= 40) return 3;
            if (pct >= 30) return 2; if (pct >= 20) return 1; return 0;
        }
        return Math.floor((score / target) * 10); // Real-time estimate
    }

    let matches = JSON.parse(localStorage.getItem('mvp_9ball_twin')) || [];

    function refresh() {
        localStorage.setItem('mvp_9ball_twin', JSON.stringify(matches));
        const container = document.getElementById('match-container');
        container.innerHTML = '';
        let tH = 0, tA = 0;

        matches.forEach((m, i) => {
            const hR = raceReq[m.h.sl], aR = raceReq[m.a.sl];
            const hPts = getMatchPoints(m.h.balls, hR, m.a.balls, aR);
            const aPts = getMatchPoints(m.a.balls, aR, m.h.balls, hR);
            tH += hPts; tA += aPts;

            const card = document.createElement('div');
            card.className = 'match-card card-active';
            card.innerHTML = `
                <div class="head-active">M${i+1} LIVE (RACE ${hR}-${aR}) 
                   <button onclick="remM(${i})" style="background:none; border:none; color:#000; font-weight:bold;">X</button>
                </div>
                <div class="lag-box">
                    LAG WINNER: 
                    <label><input type="radio" name="l${i}" ${m.lag=='h'?'checked':''} onclick="matches[${i}].lag='h';refresh()"> H</label>
                    <label><input type="radio" name="l${i}" ${m.lag=='a'?'checked':''} onclick="matches[${i}].lag='a';refresh()"> A</label>
                </div>
                <div class="grid-main">
                    ${renderSide(i,'h',m.h)}
                    ${renderSide(i,'a',m.a)}
                </div>
                <div class="inn-row">
                    INN: <button class="btn" onclick="adjM(${i},'inn',-1)">-</button> <b>${m.inn}</b> <button class="btn" onclick="adjM(${i},'inn',1)">+</button>
                </div>
                <div class="pts-row"><div class="pts-pill">PTS: ${hPts}</div><div class="pts-pill">PTS: ${aPts}</div></div>
            `;
            container.appendChild(card);
        });
        document.getElementById('h-total').innerText = tH;
        document.getElementById('a-total').innerText = tA;
        document.getElementById('h-team-label').innerText = document.getElementById('h-team').value || 'HOME';
        document.getElementById('a-team-label').innerText = document.getElementById('a-team').value || 'AWAY';
    }

    function renderSide(i,s,d) {
        return `<div class="side">
            <input type="text" class="name-input" placeholder="Player" value="${d.name}" oninput="matches[${i}].${s}.name=this.value">
            <div style="margin-bottom:5px;"><span class="sl-badge">SL ${d.sl}</span> 
            <select onchange="upSL(${i},'${s}',this.value)">${[1,2,3,4,5,6,7,8,9].map(n=>`<option ${d.sl==n?'selected':''}>${n}</option>`)}</select></div>
            <div class="stat-grid">
                <div class="stat-box"><span class="stat-label">BALLS</span><div class="stepper"><button class="btn" onclick="adj(${i},'${s}','balls',-1)">-</button><b>${d.balls}</b><button class="btn" onclick="adj(${i},'${s}','balls',1)">+</button></div></div>
            </div>
            <div class="mini-stats">
                <div class="stat-box"><span class="stat-label">DEF</span><div class="stepper"><button class="btn" onclick="adj(${i},'${s}','def',-1)">-</button><b>${d.def}</b><button class="btn" onclick="adj(${i},'${s}','def',1)">+</button></div></div>
                <div class="stat-box"><span class="stat-label">T.O</span><div class="stepper"><button class="btn" onclick="adj(${i},'${s}','to',-1)">-</button><b>${d.to}</b><button class="btn" onclick="adj(${i},'${s}','to',1)">+</button></div></div>
            </div>
        </div>`;
    }

    function adj(i,s,f,v) { matches[i][s][f] = Math.max(0, matches[i][s][f]+v); refresh(); }
    function adjM(i,f,v) { matches[i][f] = Math.max(0, matches[i][f]+v); refresh(); }
    function upSL(i,s,v) { matches[i][s].sl = parseInt(v); refresh(); }
    function remM(i) { if(confirm("Remove match?")) { matches.splice(i,1); refresh(); } }
    function addMatch() { if(matches.length<5){matches.push({inn:0,lag:'',h:{name:'',sl:3,balls:0,def:0,to:0},a:{name:'',sl:3,balls:0,def:0,to:0}}); refresh();}}
    function resetSession() { if(confirm("Reset all data?")) { matches=[]; localStorage.clear(); location.reload(); }}
    function saveT(){ localStorage.setItem('hT9', document.getElementById('h-team').value); localStorage.setItem('aT9', document.getElementById('a-team').value); refresh(); }
    function exportJPG(){ html2canvas(document.getElementById('capture-area'), {backgroundColor: "#1a1a1a"}).then(c=>{const l=document.createElement('a'); l.download='9Ball_Score.jpg'; l.href=c.toDataURL(); l.click();});}

    document.getElementById('h-team').value = localStorage.getItem('hT9') || '';
    document.getElementById('a-team').value = localStorage.getItem('aT9') || '';
    if(matches.length==0) addMatch();
    refresh();
</script>
